package com.example.sequencedice;

import java.io.Serializable;
import java.util.*;

public class Game implements Serializable {
    private final Board board;
    private final List players;
    private final List teams;
    private final Dice dice;
    private final int sequenceLength;
    private int currentPlayerIndex;
    private Player winner;
    private Team winningTeam;
    private int[] lastRoll;
    private int currentRollValue;
    private boolean waitingForPlayerMove;
    private final List observers = new ArrayList<>();

    // New fields for game statistics
    private int roundsPlayed = 0;
    private int tokensPlaced = 0;
    private int tokensRemoved = 0;

    public Game(List players, boolean teamMode) {
        if (players.size() < 2 || players.size() > 4) {
            throw new IllegalArgumentException("Player count must be 2-4");
        }
        if (players.size() == 4 && !teamMode) {
            throw new IllegalArgumentException("4 players must play in team mode");
        }
        this.players = players;
        this.board = new Board(6);
        this.dice = new Dice(2, this);
        this.teams = new ArrayList<>();
        this.sequenceLength = (players.size() == 2) ? 5 : 6;

        if (teamMode) {
            setupTeams();
        }

        List orderedPlayers = determinePlayerOrder(new ArrayList<>(players), teamMode);
        players.clear();
        players.addAll(orderedPlayers);

        this.currentPlayerIndex = 0;
        this.waitingForPlayerMove = false;

        notifyGameStarted();
    }

    public void addObserver(GameObserver observer) {
        observers.add(observer);
    }

    public void removeObserver(GameObserver observer) {
        observers.remove(observer);
    }

    private void notifyGameStarted() {
        for (GameObserver observer : observers) {
            observer.onGameStarted(
                    Collections.unmodifiableList(players),
                    teams.isEmpty() ? null : Collections.unmodifiableList(teams)
            );
        }
    }

    private void notifyPlayerTurnChanged() {
        for (GameObserver observer : observers) {
            observer.onPlayerTurnChanged(getCurrentPlayer());
        }
    }

    public void notifyDiceRolled(Player player, int[] roll, int total) {
        for (GameObserver observer : observers) {
            observer.onDiceRolled(player, roll, total);
        }
    }

    private void notifyTokenPlaced(int row, int col) {
        for (GameObserver observer : observers) {
            observer.onTokenPlaced(getCurrentPlayer(), row, col);
        }
    }

    private void notifyExtraTurnGranted() {
        for (GameObserver observer : observers) {
            observer.onExtraTurnGranted(getCurrentPlayer());
        }
    }

    private void notifyGameWon() {
        for (GameObserver observer : observers) {
            if (winner != null) {
                observer.onGameWon(winner);
            } else if (winningTeam != null) {
                observer.onTeamGameWon(winningTeam);
            }
        }
    }

    public void notifyTieDetected() {
        for (GameObserver observer : observers) {
            observer.onTieDetected();
        }
    }

    public void notifyPlayerOrderDetermined(List orderedPlayers) {
        for (GameObserver observer : observers) {
            observer.onPlayerOrderDetermined(Collections.unmodifiableList(orderedPlayers));
        }
    }


    private void setupTeams() {
        Team teamA = new Team("Team " + players.get(0).getToken().getColor());
        Team teamB = new Team("Team " + players.get(1).getToken().getColor());

        teamA.addMember(players.get(0));
        teamA.addMember(players.get(2));
        teamB.addMember(players.get(1));
        teamB.addMember(players.get(3));

        teams.add(teamA);
        teams.add(teamB);
    }
    private List determinePlayerOrder(List playerList, boolean teamMode) {
        Map playerRolls = new HashMap<>();
        Map teamRolls = new HashMap<>();

        rollForPlayers(playerList, playerRolls, teamMode, teamRolls);

        Player startingPlayer = teamMode
                ? determineTeamModeStart(playerList, playerRolls, teamRolls)
                : determineFreeForAllStart(playerList, playerRolls);

        List ordered = arrangeClockwise(playerList, startingPlayer);
        notifyPlayerOrderDetermined(ordered);
        return ordered;
    }

    private void rollForPlayers(List targetPlayers,
                                Map playerRolls,
                                boolean teamMode,
                                Map teamRolls) {
        for (Player player : targetPlayers) {
            int[] roll = dice.roll();
            int total = Arrays.stream(roll).sum();
            player.setInitialRoll(total);
            playerRolls.put(player, total);
            notifyDiceRolled(player, roll, total);

            if (teamMode) {
                Team team = player.getTeam();
                teamRolls.put(team, teamRolls.getOrDefault(team, 0) + total);
            }
        }
    }

    private Player determineTeamModeStart(List targetPlayers,
                                          Map playerRolls,
                                          Map teamRolls) {
        int highestTeamRoll = Collections.max(teamRolls.values());
        List topTeams = new ArrayList<>();
        for (Map.Entry e : teamRolls.entrySet()) {
            if (e.getValue() == highestTeamRoll) {
                Team key = e.getKey();
                topTeams.add(key);
            }
        }

        if (topTeams.size() > 1) {
            notifyTieDetected();
            return determinePlayerOrder(targetPlayers, true).get(0); // re-roll tied teams
        }

        Team winningTeam = topTeams.get(0);
        int highestPlayerRoll = winningTeam.getMembers().stream()
                .mapToInt(playerRolls::get)
                .max()
                .orElseThrow();

        List topPlayers = new ArrayList<>();
        for (Player p : winningTeam.getMembers()) {
            if (playerRolls.get(p) == highestPlayerRoll) {
                topPlayers.add(p);
            }
        }

        if (topPlayers.size() > 1) {
            notifyTieDetected();
            return determinePlayerOrder(targetPlayers, true).get(0); // re-roll tied players within team
        }

        return topPlayers.get(0);
    }

    private Player determineFreeForAllStart(List targetPlayers,
                                            Map playerRolls) {
        int highestRoll = Collections.max(playerRolls.values());
        List topPlayers = new ArrayList<>();
        for (Player p : targetPlayers) {
            if (playerRolls.get(p) == highestRoll) {
                topPlayers.add(p);
            }
        }

        if (topPlayers.size() > 1) {
            notifyTieDetected();
            return determinePlayerOrder(targetPlayers, false).get(0); // re-roll tied players
        }

        return topPlayers.get(0);
    }

    private List arrangeClockwise(List list, Player startingPlayer) {
        int startIndex = list.indexOf(startingPlayer);
        List ordered = new ArrayList<>(list.size());
        for (int i = 0; i < list.size(); i++) {
            ordered.add(list.get((startIndex + i) % list.size()));
        }
        return ordered;
    }


    public void rollDice() {
        if (isGameOver()) {
            throw new IllegalStateException("Game has already ended");
        }
        if (waitingForPlayerMove) {
            throw new IllegalStateException("Waiting for player move");
        }

        this.lastRoll = dice.roll();
        this.currentRollValue = Arrays.stream(lastRoll).sum();
        this.waitingForPlayerMove = true;

        roundsPlayed++;

        // Notify for UI display/animation
        notifyDiceRolled(getCurrentPlayer(), lastRoll, currentRollValue);
    }

    public boolean makeMove(int row, int col) {
        Player currentPlayer = getCurrentPlayer();

        if (currentRollValue == 10 && !hasOpponentTokens(currentPlayer)) {
            waitingForPlayerMove = false;
            advanceToNextPlayer();
            return false;
        }

        if (!waitingForPlayerMove || !isValidMove(row, col)) {
            return false;
        }

        Cell cell = board.getCell(row, col);

        if (currentRollValue == 10) {
            cell.setToken(null);
            tokensRemoved++;
        } else {
            cell.setToken(currentPlayer.getToken());
            tokensPlaced++;
        }

        notifyTokenPlaced(row, col);

        if (checkWinCondition()) {
            notifyGameWon();
        } else {
            if (currentRollValue == 2 || currentRollValue == 12) {
                notifyExtraTurnGranted();
            } else {
                advanceToNextPlayer();
            }
        }

        waitingForPlayerMove = false;
        return true;
    }

    // New method to check if any valid moves exist for the current roll
    public List getValidMoves() {
        List validMoves = new ArrayList<>();
        for (int r = 0; r < board.getSize(); r++) {
            for (int c = 0; c < board.getSize(); c++) {
                if (isValidMove(r, c)) {
                    validMoves.add(board.getCell(r, c));
                }
            }
        }
        return validMoves;
    }


    private boolean hasOpponentTokens(Player player) {
        String teamColor = player.isInTeam() ? player.getTeam().getTeamColor()
                : player.getToken().getColor();
        for (int r = 0; r < board.getSize(); r++) {
            for (int c = 0; c < board.getSize(); c++) {
                Cell cell = board.getCell(r, c);
                if (!cell.isEmpty() && !cell.getToken().getColor().equals(teamColor)) {
                    return true;
                }
            }
        }
        return false;
    }

    public boolean isValidMove(int row, int col) {
        if (!waitingForPlayerMove) {
            return false;
        }

        Cell cell = board.getCell(row, col);
        Player currentPlayer = getCurrentPlayer();
        String currentColor = currentPlayer.getToken().getColor();
        String teamColor = currentPlayer.isInTeam() ? currentPlayer.getTeam().getTeamColor() : currentColor;

        if (currentRollValue == 11) {
            return cell.isEmpty();
        }
        if (currentRollValue == 10) {
            return !cell.isEmpty() && !cell.getToken().getColor().equals(teamColor) && !cell.isSpecial();
        }

        if (cell.getValue() != currentRollValue) {
            return false;
        }

        boolean hasEmpty = false;
        for (int r = 0; r < board.getSize(); r++) {
            for (int c = 0; c < board.getSize(); c++) {
                Cell temp = board.getCell(r, c);
                if (temp.getValue() == currentRollValue && temp.isEmpty()) {
                    hasEmpty = true;
                    break;
                }
            }
            if (hasEmpty) break;
        }

        if (cell.isEmpty()) {
            return true;
        }

        if (!cell.getToken().getColor().equals(teamColor)) {
            return !hasEmpty;
        }

        return false;
    }

    private boolean checkWinCondition() {
        Player currentPlayer = getCurrentPlayer();
        SequenceValidator validator = new SequenceValidator(board, sequenceLength);

        if (currentPlayer.isInTeam()) {
            if (validator.hasSequence(currentPlayer.getTeam().getTeamColor())) {
                winningTeam = currentPlayer.getTeam();
                return true;
            }
        } else {
            if (validator.hasSequence(currentPlayer.getToken().getColor())) {
                winner = currentPlayer;
                return true;
            }
        }
        return false;
    }

    public void advanceToNextPlayer() {
        currentPlayerIndex = (currentPlayerIndex + 1) % players.size();
        notifyPlayerTurnChanged();
    }

    public int getRoundsPlayed() { return roundsPlayed; }
    public int getTokensPlaced() { return tokensPlaced; }
    public int getTokensRemoved() { return tokensRemoved; }

    public Map getLongestSequences() {
        SequenceValidator validator = new SequenceValidator(board, sequenceLength);
        Map sequences = new HashMap<>();
        if (players.isEmpty()) {
            return sequences;
        }

        if (players.get(0).isInTeam()) {
            for (Team team : teams) {
                sequences.put(team.getName(), validator.getLongestSequence(team.getTeamColor()));
            }
        } else {
            for (Player player : players) {
                sequences.put(player.getName(), validator.getLongestSequence(player.getToken().getColor()));
            }
        }
        return sequences;
    }

    public void setWaitingForPlayerMove(boolean waitingForPlayerMove) {
        this.waitingForPlayerMove = waitingForPlayerMove;
    }



    public Player getCurrentPlayer() { return players.get(currentPlayerIndex); }
    public Board getBoard() { return board; }
    public boolean isGameOver() { return winner != null || winningTeam != null; }
    public Player getWinner() { return winner; }
    public Team getWinningTeam() { return winningTeam; }
    public int[] getLastRoll() { return lastRoll; }
    public int getCurrentRollValue() { return currentRollValue; }
    public boolean isWaitingForPlayerMove() { return waitingForPlayerMove; }
    public List getPlayers() { return Collections.unmodifiableList(players); }

    public int getSequenceLength() {
        return sequenceLength;
    }
}

package com.example.sequencedice;

public class Board {
    private final Cell[][] grid;
    private final int size;

    public Board(int size) {
        if (size != 6)
            throw new IllegalArgumentException("Only 6x6 boards are currently supported.");
        this.size = size;
        this.grid = initializeBoard(size);
    }

    private Cell[][] initializeBoard(int size) {
        int[][] boardNumbers = {
                {2,3,4,5,6,2},
                {6,7,8,9,7,3},
                {5,9,12,12,8,4},
                {4,8,12,12,9,5},
                {3,7,9,8,7,6},
                {2,6,5,4,3,2}
        };

        Cell[][] grid = new Cell[size][size];
        for (int row = 0; row < size; row++) {
            for (int col = 0; col < size; col++) {
                grid[row][col] = new Cell(row, col, boardNumbers[row][col]);
            }
        }
        return grid;
    }

    public Cell getCell(int row,int col) {
        if (row >= size || col >= size) {
            throw new IllegalArgumentException("The Positions out of bounds");
        }
        return grid[row][col];
    }



    public int getSize() { return size; }
}

import java.util.*;
import java.util.stream.Collectors;

public class Dice {
    private static final Random RANDOM = new Random();
    private final int diceCount;
    private final Game game;

    public Dice(int diceCount, Game game) {
        this.diceCount = diceCount;
        this.game = game;
    }

    public int[] roll() {
        int[] results = new int[diceCount];
        for (int i = 0; i < diceCount; i++) {
            results[i] = RANDOM.nextInt(6) + 1;
        }
        return results;
    }

    public List determinePlayerOrder(List players, boolean teamMode) {
        if (players.size() < 2 || players.size() > 4) {
            throw new IllegalArgumentException("Invalid player count");
        }

        // First rolling phase - all players roll
        Map playerRolls = new HashMap<>();
        Map teamRolls = new HashMap<>();

        for (Player player : players) {
            int[] roll = this.roll();
            int total = Arrays.stream(roll).sum();
            player.setInitialRoll(total);
            playerRolls.put(player, total);
            game.notifyDiceRolled(player, roll, total);

            if (teamMode) {
                Team team = player.getTeam();
                teamRolls.put(team, Math.max(teamRolls.getOrDefault(team, 0), total));
            }
        }

        // Sort players based on game mode
        List orderedPlayers = new ArrayList<>();
        if (teamMode) {
            // Sort teams by their highest roll
            List sortedTeams = teamRolls.entrySet().stream()
                    .sorted(Map.Entry.comparingByValue().reversed())
                    .map(Map.Entry::getKey)
                    .collect(Collectors.toList());

            // Check for team ties
            if (sortedTeams.size() > 1 &&
                    teamRolls.get(sortedTeams.get(0)).equals(teamRolls.get(sortedTeams.get(1)))) {
                game.notifyTieDetected();
                return determinePlayerOrder(players, teamMode); // Re-roll tied teams
            }

            // Within each team, sort players by their individual rolls
            for (Team team : sortedTeams) {
                List teamPlayers = team.getMembers().stream()
                        .sorted((p1, p2) -> Integer.compare(
                                playerRolls.get(p2),
                                playerRolls.get(p1)))
                        .collect(Collectors.toList());

                // Check for player ties within team
                if (teamPlayers.size() > 1 &&
                        playerRolls.get(teamPlayers.get(0)).equals(playerRolls.get(teamPlayers.get(1)))) {
                    game.notifyTieDetected();
                    return determinePlayerOrder(players, teamMode); // Re-roll tied players
                }

                orderedPlayers.addAll(teamPlayers);
            }
        } else {
            // Sort all players by their rolls
            orderedPlayers = playerRolls.entrySet().stream()
                    .sorted(Map.Entry.comparingByValue().reversed())
                    .map(Map.Entry::getKey)
                    .collect(Collectors.toList());

            // Check for ties between players
            for (int i = 0; i < orderedPlayers.size() - 1; i++) {
                if (playerRolls.get(orderedPlayers.get(i)).equals(
                        playerRolls.get(orderedPlayers.get(i + 1)))) {
                    game.notifyTieDetected();
                    return determinePlayerOrder(players, teamMode); // Re-roll tied players
                }
            }
        }

        game.notifyPlayerOrderDetermined(orderedPlayers);
        return orderedPlayers;
    }
}

public class SequenceValidator {
    private final Board board;
    private final int requiredLength;

    public SequenceValidator(Board board, int requiredLength) {
        this.board = board;
        this.requiredLength = requiredLength;
    }

    public boolean hasSequence(String color) {
        return getLongestSequence(color) >= requiredLength;
    }

    public int getLongestSequence(String color) {
        int maxSequence = 0;
        for (int row = 0; row < board.getSize(); row++) {
            for (int col = 0; col < board.getSize(); col++) {
                Cell cell = board.getCell(row, col);
                if (!cell.isEmpty() && isMatchingColor(cell.getToken().getColor(), color)) {
                    // Check all four primary directions from this cell
                    maxSequence = Math.max(maxSequence, checkDirection(row, col, 1, 0, color));   // Vertical
                    maxSequence = Math.max(maxSequence, checkDirection(row, col, 0, 1, color));   // Horizontal
                    maxSequence = Math.max(maxSequence, checkDirection(row, col, 1, 1, color));   // Diagonal (down-right)
                    maxSequence = Math.max(maxSequence, checkDirection(row, col, 1, -1, color));  // Anti-diagonal (down-left)
                }
            }
        }
        return maxSequence;
    }

    private int checkDirection(int startRow, int startCol, int rowInc, int colInc, String color) {
        int count = 0;
        // Count in one direction
        count += countInSingleDirection(startRow, startCol, rowInc, colInc, color);
        // Count in the opposite direction
        count += countInSingleDirection(startRow, startCol, -rowInc, -colInc, color);
        // The starting cell is counted in both directions, so we subtract one
        return count - 1;
    }

    private int countInSingleDirection(int startRow, int startCol, int rowInc, int colInc, String color) {
        int count = 0;
        int currentRow = startRow;
        int currentCol = startCol;

        while (isValidPosition(currentRow, currentCol)) {
            Cell cell = board.getCell(currentRow, currentCol);
            if (cell.isEmpty() || !isMatchingColor(cell.getToken().getColor(), color)) {
                break;
            }
            count++;
            currentRow += rowInc;
            currentCol += colInc;
        }
        return count;
    }

    private boolean isMatchingColor(String cellColor, String targetColor) {
        return cellColor.equals(targetColor);
    }

    private boolean isValidPosition(int row, int col) {
        return row >= 0 && row < board.getSize() &&
                col >= 0 && col < board.getSize();
    }
}

        ViewHolder holder;
        if (convertView == null) {
            convertView = inflater.inflate(R.layout.cell_item, parent, false);
            holder = new ViewHolder();
            holder.valueText = convertView.findViewById(R.id.cell_value);
            holder.tokenView = convertView.findViewById(R.id.cell_token);
            convertView.setTag(holder);
        } else {
            holder = (ViewHolder) convertView.getTag();
        }

        // Calculate dynamic cell size based on orientation
        int cellSize = calculateCellSize(parent);
        convertView.setLayoutParams(new GridView.LayoutParams(cellSize, cellSize));

        int row = position / board.getSize();
        int col = position % board.getSize();
        Cell cell = board.getCell(row, col);

        // Set cell value
        holder.valueText.setText(String.valueOf(cell.getValue()));

        // Set token appearance using drawables
        if (cell.isEmpty()) {
            holder.tokenView.setVisibility(View.INVISIBLE);
        } else {
            holder.tokenView.setVisibility(View.VISIBLE);
            switch (cell.getToken().getColor()) {
                case "RED":
                    holder.tokenView.setImageResource(R.drawable.token_red);
                    break;
                case "BLUE":
                    holder.tokenView.setImageResource(R.drawable.token_blue);
                    break;
                case "GREEN":
                    holder.tokenView.setImageResource(R.drawable.token_green);
                    break;
                case "YELLOW":
                    holder.tokenView.setImageResource(R.drawable.token_yellow);
                    break;
            }
        }

        // Set cell background based on game state
        updateCellBackground(convertView, cell, row, col);

        return convertView;
    }

    private int calculateCellSize(ViewGroup parent) {
        DisplayMetrics metrics = context.getResources().getDisplayMetrics();
        int orientation = context.getResources().getConfiguration().orientation;

        if (orientation == Configuration.ORIENTATION_LANDSCAPE) {
            // In landscape, use height for square cells (board is on left)
            return parent.getHeight() / board.getSize();
        } else {
            // In portrait, use width minus padding and spacing
            int gridWidth = parent.getWidth();
            int padding = parent.getPaddingLeft() + parent.getPaddingRight();
            int horizontalSpacing = ((GridView) parent).getHorizontalSpacing();
            return (gridWidth - padding - ((board.getSize() - 1) * horizontalSpacing)) / board.getSize();
        }
    }

    private void updateCellBackground(View cellView, Cell cell, int row, int col) {
        if (game.isWaitingForPlayerMove() && game.isValidMove(row, col)) {
            switch (game.getCurrentRollValue()) {
                case 10:
                    cellView.setBackgroundResource(R.drawable.defensive_move_bg);
                    break;
                case 11:
                    cellView.setBackgroundResource(R.drawable.wild_move_bg);
                    break;
                default:
                    cellView.setBackgroundResource(R.drawable.valid_move_bg);
                    break;
            }
        } else if (cell.isSpecial()) {
            cellView.setBackgroundResource(R.drawable.special_cell_bg);
        } else {
            cellView.setBackgroundResource(R.drawable.normal_cell_bg);
        }
    }

    // ViewHolder pattern for better performance
    private static class ViewHolder {
        TextView valueText;
        ImageView tokenView;
    }
}

    private void startGame() {
        if (game == null) {
            showToast("Please select a game mode first");
            return;
        }

        updateButtonStates(true);
        setupBoard();
        updateGameUI();
    }

    private void setupBoard() {
        GridView gridView = findViewById(R.id.GridGame);
        boardAdapter = new BoardAdapter(this, game.getBoard(), game);
        gridView.setAdapter(boardAdapter);

        gridView.setOnItemClickListener((parent, view, position, id) -> {
            if (game != null && !game.isGameOver() && game.isWaitingForPlayerMove()) {
                handlePlayerMove(position);
            }
        });
    }

    private void handlePlayerMove(int position) {
        int row = position / 6;
        int col = position % 6;

        if (game.makeMove(row, col)) {
            updateGameUI();

            if (game.getCurrentRollValue() == 2 || game.getCurrentRollValue() == 12) {
                showToast("Extra turn! Roll again");
            }

            if (game.isGameOver()) {
                updateLeaderboard();
                startReportActivity();
            }
        } else {
            showToast("Invalid move! Try again");
        }
    }

    private void updateLeaderboard() {
        String winnerName;
        if (teamMode) {
            winnerName = game.getWinningTeam().getName();
        } else {
            winnerName = game.getWinner().getName();
        }
        LeaderboardActivity.saveWin(this, winnerName);
    }


    @SuppressLint("DefaultLocale")
    private void startReportActivity() {
        Intent intent = new Intent(this, ReportActivity.class);
        String winnerMessage;
        if (teamMode) {
            winnerMessage = "Team " + game.getWinningTeam().getTeamColor() + " wins!";
        } else {
            winnerMessage = game.getWinner().getName() + " wins!";
        }

        intent.putExtra("WINNER_MESSAGE", winnerMessage);
        intent.putExtra("ROUNDS_PLAYED", game.getRoundsPlayed());
        intent.putExtra("TOKENS_PLACED", game.getTokensPlaced());
        intent.putExtra("TOKENS_REMOVED", game.getTokensRemoved());
        intent.putExtra("LONGEST_SEQUENCES", (Serializable) game.getLongestSequences());

        startActivity(intent);
        resetGame();
    }

    @SuppressLint("DefaultLocale")
    private void handleDiceRoll() {
        game.rollDice();
        List validMoves = game.getValidMoves();

        if (validMoves.isEmpty()) {
            Toast.makeText(this, "No valid moves. Skipping turn.", Toast.LENGTH_SHORT).show();
            game.advanceToNextPlayer();
            game.setWaitingForPlayerMove(false);
            updateGameUI();
        } else {
            animateDiceRoll(findViewById(R.id.btnrolldice), () -> {
                updateGameUI();
                int[] lastRoll = game.getLastRoll();
                int total = game.getCurrentRollValue();
                String message = getRollMessage(lastRoll, total);
                showToast(message, Toast.LENGTH_LONG);
            });
        }
    }

    @SuppressLint("DefaultLocale")
    private String getRollMessage(int[] lastRoll, int total) {
        switch (total) {
            case 10:
                return String.format("%s rolled %s = %d (Defensive)\nRemove an opponent's token (not from 2 or 12 cells)",
                        game.getCurrentPlayer().getName(), Arrays.toString(lastRoll), total);
            case 11:
                return String.format("%s rolled %s = %d (Wild)\nPlace token on any empty cell",
                        game.getCurrentPlayer().getName(), Arrays.toString(lastRoll), total);
            case 2:
            case 12:
                return String.format("%s rolled %s = %d\nPlace token and get extra turn!",
                        game.getCurrentPlayer().getName(), Arrays.toString(lastRoll), total);
            default:
                return String.format("%s rolled %s = %d\nTap a %d to place your token",
                        game.getCurrentPlayer().getName(), Arrays.toString(lastRoll), total, total);
        }
    }

    private void updateGameUI() {
        boardAdapter.notifyDataSetChanged();
        updateCurrentPlayerDisplay();
        updateRollResultDisplay();
        updateButtonStates(true);
    }

    private void updateCurrentPlayerDisplay() {
        Player current = game.getCurrentPlayer();
        TextView currentPlayerText = findViewById(R.id.current_player_text);

        @SuppressLint("DefaultLocale") String playerText = teamMode ?
                String.format("%s - %s's Turn", current.getTeam().getName(), current.getName()) :
                String.format("%s's Turn", current.getName());

        currentPlayerText.setText(playerText);
        currentPlayerText.setTextColor(getPlayerColor(current.getToken().getColor()));
    }

    private int getPlayerColor(String color) {
        switch (color) {
            case "RED": return Color.RED;
            case "BLUE": return Color.BLUE;
            case "GREEN": return Color.GREEN;
            case "YELLOW": return Color.YELLOW;
            default: return Color.BLACK;
        }
    }

    private void updateRollResultDisplay() {
        TextView rollResult = findViewById(R.id.roll_result);
        if (game.getLastRoll() != null) {
            rollResult.setText(String.format("Last roll: %s = %d",
                    Arrays.toString(game.getLastRoll()), game.getCurrentRollValue()));
        } else {
            rollResult.setText("Last roll: -");
        }
    }

    private void showGameEndDialog() {
        String winnerMessage = teamMode ?
                game.getWinningTeam().getName() + " wins!" :
                game.getWinner().getName() + " wins!";

        new AlertDialog.Builder(this)
                .setTitle("Game Over")
                .setMessage(winnerMessage)
                .setPositiveButton("New Game", (dialog, which) -> resetGame())
                .setCancelable(false)
                .show();
    }

    private void showPauseDialog() {
        new AlertDialog.Builder(this)
                .setTitle("Game Paused")
                .setMessage("What would you like to do?")
                .setPositiveButton("Resume", (dialog, which) -> dialog.dismiss())
                .setNegativeButton("New Game", (dialog, which) -> resetGame())
                .setNeutralButton("Exit", (dialog, which) -> finish())
                .show();
    }

    private void resetGame() {
        game = null;
        teamMode = false;
        resetUI();
    }

    private void resetUI() {
        updateButtonStates(false);
        ((GridView) findViewById(R.id.GridGame)).setAdapter(null);
        ((TextView) findViewById(R.id.current_player_text)).setText("Current Player: -");
        ((TextView) findViewById(R.id.roll_result)).setText("Last roll: -");
        resetModeSelectionUI();
    }

    private void updateButtonStates(boolean gameStarted) {
        findViewById(R.id.btnPlay).setEnabled(!gameStarted);
        findViewById(R.id.btn2players).setEnabled(!gameStarted);
        findViewById(R.id.btn3players).setEnabled(!gameStarted);
        findViewById(R.id.button4).setEnabled(!gameStarted);
        findViewById(R.id.btnrolldice).setEnabled(gameStarted && !game.isWaitingForPlayerMove());
        findViewById(R.id.btnPause).setEnabled(gameStarted);
    }

    private void animateDiceRoll(Button rollButton, Runnable onComplete) {
        rollButton.setEnabled(false);
        ImageView dice1 = findViewById(R.id.dice1_image_view);
        ImageView dice2 = findViewById(R.id.dice2_image_view);
        Random random = new Random();

        final int[] diceDrawables = {
                R.drawable.dice_1, R.drawable.dice_2, R.drawable.dice_3,
                R.drawable.dice_4, R.drawable.dice_5, R.drawable.dice_6
        };

        Animation anim = new RotateAnimation(0, 360,
                Animation.RELATIVE_TO_SELF, 0.5f,
                Animation.RELATIVE_TO_SELF, 0.5f);
        anim.setDuration(100);
        anim.setRepeatCount(10);
        anim.setInterpolator(new AccelerateDecelerateInterpolator());

        anim.setAnimationListener(new Animation.AnimationListener() {
            @Override
            public void onAnimationStart(Animation animation) {
            }

            @Override
            public void onAnimationEnd(Animation animation) {
                rollButton.setEnabled(true);
                int[] lastRoll = game.getLastRoll();
                if (lastRoll != null && lastRoll.length == 2) {
                    dice1.setImageResource(diceDrawables[lastRoll[0] - 1]);
                    dice2.setImageResource(diceDrawables[lastRoll[1] - 1]);
                }
                onComplete.run();
            }

            @Override
            public void onAnimationRepeat(Animation animation) {
                dice1.setImageResource(diceDrawables[random.nextInt(6)]);
                dice2.setImageResource(diceDrawables[random.nextInt(6)]);
            }
        });

        dice1.startAnimation(anim);
        dice2.startAnimation(anim);
    }

    private void showToast(String message) {
        Toast.makeText(this, message, Toast.LENGTH_SHORT).show();
    }

    private void showToast(String message, int duration) {
        Toast.makeText(this, message, duration).show();
    }
}
