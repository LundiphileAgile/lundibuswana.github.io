Sequence Dice Game - Source Code

package com.example.sequencedice;

import java.util.*;
import java.io.Serializable;

public class Game implements Serializable {
    private final Board board;
    private final List players;
    private final List teams;
    private final Dice dice;
    private final int sequenceLength;
    private int currentPlayerIndex;
    private Player winner;
    private Team winningTeam;
    private int[] lastRoll;
    private int currentRollValue;
    private boolean waitingForPlayerMove;
    private final List observers = new ArrayList<>();

    // Game statistics
    private int roundsPlayed = 0;
    private int tokensPlaced = 0;
    private int tokensRemoved = 0;

    public Game(List players, boolean teamMode) {
        if (players.size() < 2 || players.size() > 4) {
            throw new IllegalArgumentException("Player count must be 2-4");
        }
        if (players.size() == 4 && !teamMode) {
            throw new IllegalArgumentException("4 players must play in team mode");
        }
        
        this.players = players;
        this.board = new Board(6);
        this.dice = new Dice(2, this);
        this.teams = new ArrayList<>();
        this.sequenceLength = (players.size() == 2) ? 5 : 6;

        if (teamMode) {
            setupTeams();
        }

        List orderedPlayers = determinePlayerOrder(new ArrayList<>(players), teamMode);
        players.clear();
        players.addAll(orderedPlayers);

        this.currentPlayerIndex = 0;
        this.waitingForPlayerMove = false;

        notifyGameStarted();
    }

    private void setupTeams() {
        Team teamA = new Team("Team " + players.get(0).getToken().getColor());
        Team teamB = new Team("Team " + players.get(1).getToken().getColor());

        teamA.addMember(players.get(0));
        teamA.addMember(players.get(2));
        teamB.addMember(players.get(1));
        teamB.addMember(players.get(3));

        teams.add(teamA);
        teams.add(teamB);
    }

    public void rollDice() {
        if (isGameOver()) {
            throw new IllegalStateException("Game has already ended");
        }
        if (waitingForPlayerMove) {
            throw new IllegalStateException("Waiting for player move");
        }

        this.lastRoll = dice.roll();
        this.currentRollValue = Arrays.stream(lastRoll).sum();
        this.waitingForPlayerMove = true;

        roundsPlayed++;
        notifyDiceRolled(getCurrentPlayer(), lastRoll, currentRollValue);
    }

    public boolean makeMove(int row, int col) {
        Player currentPlayer = getCurrentPlayer();

        if (currentRollValue == 10 && !hasOpponentTokens(currentPlayer)) {
            waitingForPlayerMove = false;
            advanceToNextPlayer();
            return false;
        }

        if (!waitingForPlayerMove || !isValidMove(row, col)) {
            return false;
        }

        Cell cell = board.getCell(row, col);

        if (currentRollValue == 10) {
            cell.setToken(null);
            tokensRemoved++;
        } else {
            cell.setToken(currentPlayer.getToken());
            tokensPlaced++;
        }

        notifyTokenPlaced(row, col);

        if (checkWinCondition()) {
            notifyGameWon();
        } else {
            if (currentRollValue == 2 || currentRollValue == 12) {
                notifyExtraTurnGranted();
            } else {
                advanceToNextPlayer();
            }
        }

        waitingForPlayerMove = false;
        return true;
    }

    public boolean isValidMove(int row, int col) {
        if (!waitingForPlayerMove) {
            return false;
        }

        Cell cell = board.getCell(row, col);
        Player currentPlayer = getCurrentPlayer();
        String currentColor = currentPlayer.getToken().getColor();
        String teamColor = currentPlayer.isInTeam() ? currentPlayer.getTeam().getTeamColor() : currentColor;

        if (currentRollValue == 11) {
            return cell.isEmpty();
        }
        if (currentRollValue == 10) {
            return !cell.isEmpty() && !cell.getToken().getColor().equals(teamColor) && !cell.isSpecial();
        }

        if (cell.getValue() != currentRollValue) {
            return false;
        }

        boolean hasEmpty = false;
        for (int r = 0; r < board.getSize(); r++) {
            for (int c = 0; c < board.getSize(); c++) {
                Cell temp = board.getCell(r, c);
                if (temp.getValue() == currentRollValue && temp.isEmpty()) {
                    hasEmpty = true;
                    break;
                }
            }
            if (hasEmpty) break;
        }

        if (cell.isEmpty()) {
            return true;
        }

        if (!cell.getToken().getColor().equals(teamColor)) {
            return !hasEmpty;
        }

        return false;
    }

    // Getters
    public Player getCurrentPlayer() { return players.get(currentPlayerIndex); }
    public Board getBoard() { return board; }
    public boolean isGameOver() { return winner != null || winningTeam != null; }
    public Player getWinner() { return winner; }
    public Team getWinningTeam() { return winningTeam; }
    public int[] getLastRoll() { return lastRoll; }
    public int getCurrentRollValue() { return currentRollValue; }
    public boolean isWaitingForPlayerMove() { return waitingForPlayerMove; }
    public List getPlayers() { return Collections.unmodifiableList(players); }
}

package com.example.sequencedice;

public class Board {
    private final Cell[][] grid;
    private final int size;

    public Board(int size) {
        if (size != 6)
            throw new IllegalArgumentException("Only 6x6 boards are currently supported.");
        this.size = size;
        this.grid = initializeBoard(size);
    }

    private Cell[][] initializeBoard(int size) {
        int[][] boardNumbers = {
                {2,3,4,5,6,2},
                {6,7,8,9,7,3},
                {5,9,12,12,8,4},
                {4,8,12,12,9,5},
                {3,7,9,8,7,6},
                {2,6,5,4,3,2}
        };

        Cell[][] grid = new Cell[size][size];
        for (int row = 0; row < size; row++) {
            for (int col = 0; col < size; col++) {
                grid[row][col] = new Cell(row, col, boardNumbers[row][col]);
            }
        }
        return grid;
    }

    public Cell getCell(int row,int col) {
        if (row >= size || col >= size) {
            throw new IllegalArgumentException("The Positions out of bounds");
        }
        return grid[row][col];
    }

    public int getSize() { return size; }
}
package com.example.sequencedice;

public class Cell {
    private final int row;
    private final int col;
    private final int value;
    private Token token;

    public Cell(int row,int col, int value) {
        this.row=row;
        this.col=col;
        this.value = value;
    }

    public int getValue() {
        return value;
    }
    public Token getToken() {
        return token;
    }
    public void setToken(Token token) {
        this.token = token;
    }
    public boolean isEmpty() {
        return token == null;
    }

    public boolean isSpecial() {
        return value == 2 || value == 12; // Extra turn cells
    }
}
